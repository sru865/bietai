//SWAPPING of 2 numbers using 3rd variable
#include<stdio.h>
int main()
{
int a=10, b=20,c;
printf("Before swapping A=%d B=%d",a,b);
c=a;
a=b;
b=c;
printf("After swapping A=%d B=%d",a,b);
}
//same program using scanf
#include<stdio.h>
print("Enter the value of A and B\n");
scanf("%d%d",&a,&b);
c=a+b;
printf("RESULT=%d",c);
}
